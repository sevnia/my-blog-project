### DB : oracle 21

##### 테이블 명 : blogboard (귀찮아서 코드 복붙)
CREATE TABLE blogboard(  
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title VARCHAR2(200 CHAR),
    writer VARCHAR2(100 CHAR),
    blogcontent VARCHAR2(2000 CHAR) NOT NULL
);
